#Python ap
import requests
# Prepare URL
occ = "http://purl.org/nhmuio/id/5cbe9ccf-4013-4f45-b22b-0d2b0f839d59"
url = f'https://api.gbif.org/v1/occurrence/search?q={occ}'
# Make API call
response = requests.get(url)
# Get status code
print(response.status_code)
# Access data
print(response.json())
# Get all results
all = response.json()["results"]
# Get first record in results
first = response.json()["results"][0]
# Get specific value in record
first["materialSampleID"]
# Loop over all results to extract a specific ID
[i["materialSampleID"] for i in all]
